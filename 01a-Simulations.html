library(glue)
rolls <- c()
for(j in 1:16){
  rolls <- c()
  for(i in 1:10000){
    rolls <- c(rolls,play(board,state=j))
  }
  m <- round(mean(rolls),2)
  s <- round(sd(rolls)^2,2)
  hist(rolls,breaks=50,xlim=c(0, 50),main=glue("State {j} Start: # of Rolls to Win."))
  mtext(glue(" Mean = {m}. Var = {s}."), side=3, line = - 2)
  abline(v=m, col='red')
}
